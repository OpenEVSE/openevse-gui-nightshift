# Schedule Screen Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the v3 Schedule screen — the `/schedule` route — a list of charge timers with add/edit/delete, replacing the placeholder screen.

**Architecture:** A pure-logic module (`schedule/timers.js`) handles day/flag conversion, id assignment, and validation. Four Schedule components render the screen from plain props. `Schedule.svelte` is the only store-aware unit: it reads `schedule_store`, composes the components, and routes add/edit/delete callbacks back to the store through `serialQueue`. Reuses existing UI primitives — no new primitive needed.

**Tech Stack:** Svelte 5 (runes), Tailwind 4 (CSS-variable theme tokens), `svelte-i18n`, Vitest + `@testing-library/svelte`.

**Preconditions:**
- The v3 foundation + Dashboard are merged to `main`. Work happens on a `schedule` branch (the executor creates it).
- 240 tests pass. UI primitives available: `Card`, `Button`, `IconButton`, `Icon`, `Modal`, `SegmentedControl`. Stores: `schedule`, `uistates`. Helpers: `displayTime` in `src/lib/utils.js`. `serialQueue` in `src/lib/queue.js`.

**Data model:** `schedule_store` holds an array of timer objects: `{ id: int, state: "active"|"disabled", time: "HH:MM", days: string[] }` where `days` are lowercase Monday-first English day names. Store methods: `download()` → `GET /schedule`; `upload(timer)` → `POST /schedule`; `remove(id)` → `DELETE /schedule/<id>`. All return a boolean.

**Plan-level decisions:**
- The route re-downloads `schedule_store` after a successful write (single source of truth) rather than locally mutating the array.
- The `TimerModal` is id-agnostic — it emits `{state, time, days}`; the route assigns `id` (new) or merges into the existing timer (edit).
- Failed writes surface the global `AlertBox` via `uistates_store.setObject('alertbox', …)` (wired in `App.svelte`).
- Components never import stores; `Schedule.svelte` passes data down and callbacks up.

```js
/** Pure helpers for the Schedule screen. No store or DOM access — fully unit-tested. */

/** The seven weekdays in the device's order (Monday-first). Single source of truth. */
export const DAYS = ['monday', 'tuesday', 'wednesday', 'thursday', 'friday', 'saturday', 'sunday']

/** Next free timer id: highest existing id + 1, or 1 when the list is empty. */
export function nextTimerId(timers) {
  if (!Array.isArray(timers) || timers.length === 0) return 1
  let max = 0
  for (const t of timers) {
    const id = Number(t?.id)
    if (Number.isFinite(id) && id > max) max = id
  }
  return max + 1
}

/** A `days` name array → a Monday-first boolean[7]. */
export function daysToFlags(days) {
  const set = new Set(Array.isArray(days) ? days : [])
  return DAYS.map((d) => set.has(d))
}

/** A Monday-first boolean[7] → a `days` name array. */
export function flagsToDays(flags) {
  const f = Array.isArray(flags) ? flags : []
  return DAYS.filter((_, i) => f[i] === true)
}

/** True when at least one day flag is set (the save-validation rule). */
export function hasAnyDay(flags) {
  return Array.isArray(flags) && flags.some((x) => x === true)
}

/** All seven days as `{ day, on }`, in order, for the TimerRow chip strip. */
export function formatDayChips(days) {
  const set = new Set(Array.isArray(days) ? days : [])
  return DAYS.map((d) => ({ day: d, on: set.has(d) }))
}
```

```svelte
<script>
  import { _ } from 'svelte-i18n'
  import { DAYS } from '../../schedule/timers.js'

  let { flags = [false, false, false, false, false, false, false], onchange = () => {} } = $props()

  let allOn = $derived(flags.every((f) => f === true))

  function toggle(i) {
    const next = flags.slice()
    next[i] = !next[i]
    onchange(next)
  }
  function toggleAll() {
    onchange(DAYS.map(() => !allOn))
  }
</script>

<div class="flex justify-between gap-1">
  {#each DAYS as day, i}
    <button
      type="button"
      aria-pressed={flags[i] === true}
      aria-label={$_('days.' + day)}
      onclick={() => toggle(i)}
      class="flex-1 rounded-lg py-2 text-[11px] font-semibold transition
             {flags[i] ? 'bg-accent text-surface' : 'bg-surface-3 text-text-dim'}"
    >
      {$_('days.' + day).slice(0, 3)}
    </button>
  {/each}
</div>
<button
  type="button"
  onclick={toggleAll}
  class="mt-2 text-xs font-semibold text-accent"
>
  {allOn ? $_('schedule.clear_all') : $_('schedule.select_all')}
</button>
```

```svelte
<script>
  import { _ } from 'svelte-i18n'
  import Modal from '../ui/Modal.svelte'
  import Button from '../ui/Button.svelte'
  import SegmentedControl from '../ui/SegmentedControl.svelte'
  import DayPicker from './DayPicker.svelte'
  import { daysToFlags, flagsToDays, hasAnyDay, DAYS } from '../../schedule/timers.js'

  let { open = false, timer = null, busy = false, onclose = () => {}, onsave = () => {} } = $props()

  let flags = $state(DAYS.map(() => true))
  let time = $state('08:00')
  let timerState = $state('active')
  let showDayError = $state(false)

  $effect(() => {
    if (open) {
      flags = timer ? daysToFlags(timer.days) : DAYS.map(() => true)
      time = timer?.time ?? '08:00'
      timerState = timer?.state ?? 'active'
      showDayError = false
    }
  })

  let stateOptions = $derived([
    { value: 'active', label: $_('schedule.active') },
    { value: 'disabled', label: $_('schedule.disabled') },
  ])

  function save() {
    if (!hasAnyDay(flags)) {
      showDayError = true
      return
    }
    onsave({ state: timerState, time, days: flagsToDays(flags) })
  }
</script>

<Modal visible={open} closable={!busy} {onclose}>
  <h2 class="mb-4 text-base font-semibold text-text">
    {timer ? $_('schedule.edit_title') : $_('schedule.new_title')}
  </h2>

  <DayPicker {flags} onchange={(f) => { flags = f; showDayError = false }} />
  {#if showDayError}
    <p class="mt-2 text-xs text-error">{$_('schedule.error_no_day')}</p>
  {/if}

  <label class="mt-4 block">
    <span class="mb-1 block text-[10px] tracking-wide text-text-dim uppercase">{$_('schedule.time')}</span>
    <input
      type="time"
      bind:value={time}
      class="w-full rounded-xl border border-border bg-surface-2 px-3 py-2 text-sm text-text"
    />
  </label>

  <div class="mt-4">
    <span class="mb-1 block text-[10px] tracking-wide text-text-dim uppercase">{$_('schedule.state')}</span>
    <SegmentedControl options={stateOptions} value={timerState} onchange={(v) => (timerState = v)} />
  </div>

  <div class="mt-5 flex gap-2">
    <Button label={$_('schedule.save')} disabled={busy} onclick={save} />
    <Button label={$_('schedule.cancel')} variant="ghost" disabled={busy} onclick={onclose} />
```

```svelte
<script>
  import { _ } from 'svelte-i18n'
  import { schedule_store } from '../lib/stores/schedule.js'
  import { uistates_store } from '../lib/stores/uistates.js'
  import { serialQueue } from '../lib/queue.js'
  import { nextTimerId } from '../lib/schedule/timers.js'
  import Button from '../lib/components/ui/Button.svelte'
  import TimerList from '../lib/components/schedule/TimerList.svelte'
  import TimerModal from '../lib/components/schedule/TimerModal.svelte'

  const MAX_TIMERS = 50

  let editorOpen = $state(false)
  let editingTimer = $state(null)
  let busy = $state(false)
  let removingId = $state(null)

  let timers = $derived(Array.isArray($schedule_store) ? $schedule_store : [])

  function alertFail() {
    uistates_store.setObject('alertbox', {
      title: $_('schedule.error_title'),
      body: $_('schedule.error_body'),
      visible: true,
      button: true,
      closable: true,
      component: undefined,
      action: () => uistates_store.resetAlertBox(),
    })
  }

  function openAdd() {
    editingTimer = null
    editorOpen = true
  }
  function openEdit(timer) {
    editingTimer = timer
    editorOpen = true
  }

  async function save(data) {
    if (busy) return
    busy = true
    try {
      const timer = editingTimer
        ? { ...editingTimer, ...data }
        : { id: nextTimerId(timers), ...data }
      const ok = await serialQueue.add(() => schedule_store.upload(timer))
      if (ok) {
        editorOpen = false
        await serialQueue.add(() => schedule_store.download())
      } else {
        alertFail()
      }
    } finally {
      busy = false
    }
  }

  async function remove(id) {
    if (busy) return
    busy = true
    removingId = id
    try {
      const ok = await serialQueue.add(() => schedule_store.remove(id))
      if (ok) {
        await serialQueue.add(() => schedule_store.download())
      } else {
        alertFail()
      }
    } finally {
      busy = false
      removingId = null
    }
  }
</script>

<section class="p-4">
  <div class="mb-3 flex items-baseline justify-between">
    <h1 class="text-lg font-semibold text-text">{$_('screen.schedule')}</h1>
    <span class="text-xs text-text-dim">{timers.length} / {MAX_TIMERS}</span>
  </div>

  <TimerList
    {timers}
    {removingId}
    disabled={busy}
    onedit={openEdit}
    ondelete={remove}
  />

  <div class="mt-4">
    <Button
      label={'+ ' + $_('schedule.new')}
      disabled={busy || timers.length >= MAX_TIMERS}
      onclick={openAdd}
    />
  </div>
</section>

<TimerModal
  open={editorOpen}
  timer={editingTimer}
  {busy}
  onclose={() => (editorOpen = false)}
  onsave={save}
/>
```
